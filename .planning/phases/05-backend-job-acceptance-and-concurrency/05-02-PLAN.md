    ```typescript
    import { NextResponse } from 'next/server'
    import { db } from '../../../../lib/db'
    import { jobs } from '../../../../lib/db/schema'
    import { getAuthenticatedUser } from '../../../../lib/auth'
    import { JobDto, ApiSuccessResponse, JobStatus } from '@local/types'
    import { eq, and } from 'drizzle-orm'

    // ... existing POST handler code ...

    export async function GET(req: Request) {
      // 1. Authenticate user (optional for browsing jobs, but good practice)
      const user = await getAuthenticatedUser(req)
      if (!user) {
        return NextResponse.json({ errors: { auth: 'unauthorized' } }, { status: 401 })
      }

      // 2. Parse query parameters
      const url = new URL(req.url)
      const cityArea = url.searchParams.get('cityArea')
      const status = url.searchParams.get('status') as JobStatus | null
      const includeAccepted = url.searchParams.get('includeAccepted') === 'true'

      // 3. Build WHERE conditions
      const conditions = []

      // Filter by cityArea if provided (for providers browsing jobs in their area)
      if (cityArea) {
        conditions.push(eq(jobs.cityArea, cityArea))
      }

      // Filter by status if provided
      if (status) {
        conditions.push(eq(jobs.status, status))
      } else if (!includeAccepted) {
        // Default: only show PENDING jobs (accepted jobs shouldn't appear in open list)
        conditions.push(eq(jobs.status, JobStatus.PENDING))
      }

      // 4. Execute query
      const jobList = await db
        .select()
        .from(jobs)
        .where(conditions.length > 0 ? and(...conditions) : undefined)
        .orderBy(jobs.createdAt)

      // 5. Map to JobDto array
      const jobDtos: JobDto[] = jobList.map(job => ({
        id: String(job.id),
        status: job.status as JobStatus,
        version: job.version,
        category: job.category,
        description: job.description,
        timeframe: job.timeframe,
        cityArea: job.cityArea,
        clientId: job.clientId,
        providerId: job.providerId,
        createdAt: job.createdAt.toISOString(),
        updatedAt: job.updatedAt.toISOString(),
      }))

      return NextResponse.json({ data: jobDtos } as ApiSuccessResponse<JobDto[]>)
    }
    ```

    Key behaviors:
    - GET /api/jobs?cityArea=Springfield&status=PENDING → returns only PENDING jobs in Springfield
    - GET /api/jobs without status defaults to PENDING only (accepted jobs excluded)
    - GET /api/jobs?includeAccepted=true → includes all statuses
    - Provider browsing jobs uses: GET /api/jobs?cityArea={their_area}&status=PENDING

    ```typescript
    import { describe, it, expect, beforeAll, afterAll } from '@jest/globals'
    import { db } from '../../apps/web/lib/db'
    import { jobs } from '../../apps/web/lib/db/schema'
    import { JobStatus } from '@local/types'

    // Helper to create a test job
    async function createTestJob() {
      const [job] = await db.insert(jobs).values({
        category: 'PLUMBING',
        description: 'Test job for concurrent acceptance',
        timeframe: '2 hours',
        cityArea: 'TestCity',
        clientId: '1',
        status: JobStatus.PENDING,
        version: 1,
      }).returning()
      return job
    }

    // Helper to simulate acceptance request
    async function acceptJob(jobId: number, version: number, providerId: string) {
      // Simulate two concurrent requests using Promise.all
      const baseUrl = process.env.TEST_API_URL || '<http://localhost:3000>'
      
      const response = await fetch(`${baseUrl}/api/jobs/${jobId}/accept`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          // In real test, would include Authorization header with provider token
        },
        body: JSON.stringify({ version }),
      })
      
      return {
        status: response.status,
        body: await response.json(),
      }
    }

    describe('Concurrent Job Acceptance', () => {
      let testJob: any

      beforeAll(async () => {
        testJob = await createTestJob()
      })

      afterAll(async () => {
        // Cleanup: delete test job
        await db.delete(jobs).where(eq(jobs.id, testJob.id))
      })

      it('should allow exactly one provider to accept a job when multiple concurrent requests are made', async () => {
        // Simulate two providers trying to accept the same job concurrently
        const provider1Id = 'provider-1'
        const provider2Id = 'provider-2'
        const version = testJob.version

        // In a real test with auth, we'd set up two different provider tokens
        // For now, we test the version conflict mechanism
        
        // This test verifies the optimistic concurrency control logic
        // The actual concurrent test would require:
        // 1. Two valid provider JWTs
        // 2. Making two simultaneous POST requests to /api/jobs/:id/accept
        // 3. Verifying exactly one returns 200 and the other returns 409
        
        // Simplified single-request test:
        const result1 = await acceptJob(testJob.id, version, provider1Id)
        
        // First acceptance should succeed (200)
        expect(result1.status).toBe(200)
        expect(result1.body.data).toBeDefined()
        expect(result1.body.data.status).toBe(JobStatus.ACCEPTED)
        expect(result1.body.data.version).toBe(version + 1)
        
        // Try second acceptance with stale version
        const result2 = await acceptJob(testJob.id, version, provider2Id)
        
        // Second acceptance should fail with 409 (version conflict)
        expect(result2.status).toBe(409)
        expect(result2.body.errors).toBeDefined()
        expect(result2.body.errors.version).toBe('conflict')
      })

      it('should exclude accepted jobs from PENDING list', async () => {
        // Verify the accepted job is not in the PENDING list
        const baseUrl = process.env.TEST_API_URL || '<http://localhost:3000>'
        
        const response = await fetch(`${baseUrl}/api/jobs?cityArea=TestCity&status=PENDING`)
        const body = await response.json()
        
        const acceptedJobInList = body.data?.find((job: any) => job.id === String(testJob.id))
        expect(acceptedJobInList).toBeUndefined()
      })
    })
    ```
