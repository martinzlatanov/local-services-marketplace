    ```typescript
    import { NextResponse } from 'next/server'
    import { db } from '../../../../../../lib/db'
    import { jobs } from '../../../../../../lib/db/schema'
    import { getAuthenticatedUser } from '../../../../../../lib/auth'
    import { AcceptJobRequest, JobDto, ApiSuccessResponse, ApiErrorResponse, JobStatus } from '@local/types'
    import { eq, and } from 'drizzle-orm'

    export async function POST(req: Request, { params }: { params: { id: string } }) {
      // 1. Authenticate user
      const user = await getAuthenticatedUser(req)
      if (!user) {
        return NextResponse.json({ errors: { auth: 'unauthorized' } }, { status: 401 })
      }

      // 2. Enforce PROVIDER role
      if (user.role !== 'PROVIDER') {
        return NextResponse.json({ errors: { role: 'only_providers_can_accept_jobs' } }, { status: 403 })
      }

      // 3. Parse request body
      const body = await req.json().catch(() => null)
      if (!body) {
        return NextResponse.json({ errors: { body: 'invalid_json' } }, { status: 400 })
      }
      
      const payload = body as AcceptJobRequest
      
      // 4. Validate version is provided
      if (payload.version === undefined || payload.version === null) {
        return NextResponse.json({ errors: { version: 'required' } }, { status: 400 })
      }

      const jobId = parseInt(params.id, 10)
      if (isNaN(jobId)) {
        return NextResponse.json({ errors: { id: 'invalid_job_id' } }, { status: 400 })
      }

      // 5. Atomically update with version check (optimistic concurrency)
      // The key is: UPDATE jobs SET status='ACCEPTED', provider_id=user.id, version=version+1 
      // WHERE id=jobId AND version=payload.version AND status='PENDING'
      const [updatedJob] = await db
        .update(jobs)
        .set({
          status: JobStatus.ACCEPTED,
          providerId: String(user.id),
          version: payload.version + 1,
          updatedAt: new Date(),
        })
        .where(
          and(
            eq(jobs.id, jobId),
            eq(jobs.version, payload.version),
            eq(jobs.status, JobStatus.PENDING)
          )
        )
        .returning()

      // 6. Check if update succeeded (version mismatch or job not found)
      if (!updatedJob) {
        // Check if job exists and is not PENDING (already accepted or in progress)
        const [existingJob] = await db
          .select()
          .from(jobs)
          .where(eq(jobs.id, jobId))
          .limit(1)

        if (!existingJob) {
          return NextResponse.json({ errors: { job: 'not_found' } }, { status: 404 })
        }

        if (existingJob.status !== JobStatus.PENDING) {
          return NextResponse.json({ errors: { job: 'not_pending' } }, { status: 409 })
        }

        // Job exists and is PENDING but version didn't match → concurrent modification
        return NextResponse.json({ errors: { version: 'conflict', message: 'Job was updated by another provider' } }, { status: 409 })
      }

      // 7. Return updated JobDto
      const jobDto: JobDto = {
        id: String(updatedJob.id),
        status: updatedJob.status as JobStatus,
        version: updatedJob.version,
        category: updatedJob.category,
        description: updatedJob.description,
        timeframe: updatedJob.timeframe,
        cityArea: updatedJob.cityArea,
        clientId: updatedJob.clientId,
        providerId: updatedJob.providerId,
        createdAt: updatedJob.createdAt.toISOString(),
        updatedAt: updatedJob.updatedAt.toISOString(),
      }

      return NextResponse.json({ data: jobDto } as ApiSuccessResponse<JobDto>)
    }
    ```
