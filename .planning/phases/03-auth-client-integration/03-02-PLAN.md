  Create apps/web/contexts/AuthContext.tsx:
  ```typescript
  'use client'

  import { createContext, useContext, useEffect, useState } from 'react'
  import type { AuthUserDto } from '@local/types'

  interface AuthContextValue {
    user: AuthUserDto | null
    isLoading: boolean
    login: (email: string, password: string) => Promise<void>
    logout: () => Promise<void>
  }

  const AuthContext = createContext<AuthContextValue | null>(null)

  export function AuthProvider({ children }: { children: React.ReactNode }) {
    const [user, setUser] = useState<AuthUserDto | null>(null)
    const [isLoading, setIsLoading] = useState(true)

    useEffect(() => {
      // D-01: call /me once on mount to rehydrate session from httpOnly cookie
      fetch('/api/auth/me')
        .then((res) => (res.ok ? res.json() : null))
        .then((data) => {
          setUser(data?.user ?? null)
        })
        .catch(() => {
          setUser(null)
        })
        .finally(() => {
          setIsLoading(false)
        })
    }, [])

    async function login(email: string, password: string): Promise<void> {
      const res = await fetch('/api/auth/login', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email, password }),
      })
      if (!res.ok) {
        const data = await res.json()
        throw data // caller handles field-map errors per D-03
      }
      const data = await res.json()
      setUser(data.user)
    }

    async function logout(): Promise<void> {
      await fetch('/api/auth/logout', { method: 'POST' })
      setUser(null)
    }

    return (
      <AuthContext.Provider value={{ user, isLoading, login, logout }}>
        {children}
      </AuthContext.Provider>
    )
  }

  export function useAuth(): AuthContextValue {
    const ctx = useContext(AuthContext)
    if (!ctx) throw new Error('useAuth must be used within AuthProvider')
    return ctx
  }
  ```

  Key constraints:
  - `'use client'` MUST be the first line — this file uses useState, useEffect, useContext
  - `login()` throws the raw API error payload (field-map) so calling pages can display per-field errors per Phase 2 D-03
  - `logout()` calls /api/auth/logout then clears local state (D-04 from Phase 2: server-side invalidation + client clear)
  - `isLoading` starts `true` and becomes `false` only after /me resolves — pages use this for the loading spinner per UI-SPEC

  Per D-04: middleware checks cookie existence on protected routes and redirects unauthenticated users to /login.

  **Create apps/web/middleware.ts:**
  ```typescript
  import { NextRequest, NextResponse } from 'next/server'

  const PROTECTED_PATHS = ['/dashboard', '/jobs', '/post-job']
  const PUBLIC_AUTH_PATHS = ['/login', '/register']

  export function middleware(req: NextRequest) {
    const { pathname } = req.nextUrl
    const token = req.cookies.get('token')?.value

    const isProtected = PROTECTED_PATHS.some((p) => pathname.startsWith(p))
    const isPublicAuth = PUBLIC_AUTH_PATHS.some((p) => pathname.startsWith(p))

    if (isProtected && !token) {
      return NextResponse.redirect(new URL('/login', req.url))
    }

    if (isPublicAuth && token) {
      return NextResponse.redirect(new URL('/dashboard', req.url))
    }

    return NextResponse.next()
  }

  export const config = {
    matcher: ['/dashboard/:path*', '/jobs/:path*', '/post-job/:path*', '/login', '/register'],
  }
  ```

  Middleware checks cookie EXISTENCE only (fast path) — actual JWT verification happens in AuthContext via /me on page load per D-04.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser -> Next.js middleware | Cookie read at the Edge before any page renders; existence check is fast but does not verify JWT signature |
| AuthContext -> /api/auth/me | fetch crosses the browser-to-API boundary; response contains user payload derived from signed JWT |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04 | S | apps/web/middleware.ts | accept | Middleware checks cookie existence, not JWT validity. An attacker with any non-empty 'token' cookie value would pass the gate but /me (which verifies the signature) would return 401. AuthContext sets user to null on 401 and pages redirect to /login. Acceptable for a fast-path guard. |
| T-03-05 | E | apps/web/contexts/AuthContext.tsx | mitigate | `login()` throws raw API response — caller must not render untrusted error strings as raw HTML. Pages must use `textContent` / React text nodes (not dangerouslySetInnerHTML) when displaying error messages from the field map. |
| T-03-06 | I | apps/web/app/layout.tsx | accept | globals.css import is build-time only; no runtime injection surface. Tailwind purges unused classes at build — no sensitive class names leak via CSS. |
</threat_model>
