**Implementation outline:**
```typescript
import { useCallback, useEffect, useState } from 'react'
import { FlatList, RefreshControl, StyleSheet, View } from 'react-native'
import { useFocusEffect } from '@expo/router'
import { Appbar, Card, Button, Text, ActivityIndicator, useTheme } from 'react-native-paper'
import * as SecureStore from 'expo-secure-store'
import { JobDto } from '@local/types'
import { useAuth, TOKEN_KEY } from '../../../contexts/AuthContext'
import { getMyJobs } from '../../../lib/api'
import { useRouter } from 'expo-router'

export default function ActiveJobsScreen() {
  const { user } = useAuth()
  const router = useRouter()
  const theme = useTheme()
  const [token, setToken] = useState<string | null>(null)
  const [jobs, setJobs] = useState<JobDto[]>([])
  const [isLoading, setIsLoading] = useState(true)
  const [isRefreshing, setIsRefreshing] = useState(false)
  const [errorMessage, setErrorMessage] = useState<string | null>(null)

  // Token loading (same pattern as feed.tsx)
  useEffect(() => {
    let isActive = true
    async function loadToken() {
      const stored = await SecureStore.getItemAsync(TOKEN_KEY)
      if (isActive) setToken(stored)
    }
    if (user) void loadToken()
    return () => { isActive = false }
  }, [user])

  // Fetch function
  const fetchJobs = useCallback(
    async (refresh = false) => {
      if (!token) return
      if (refresh) setIsRefreshing(true)
      else setIsLoading(true)
      setErrorMessage(null)
      try {
        const data = await getMyJobs(token)
        setJobs(data)
      } catch (err: any) {
        setErrorMessage(err?.errors?.['job'] || "Couldn't load active jobs")
      } finally {
        setIsLoading(false)
        setIsRefreshing(false)
      }
    },
    [token]
  )

  // Initial load
  useEffect(() => {
    void fetchJobs()
  }, [token, fetchJobs])

  // Tab focus refresh (per D-04)
  useFocusEffect(
    useCallback(() => {
      void fetchJobs()
    }, [fetchJobs])
  )

  return (
    <View style={styles.container}>
      <Appbar.Header>
        <Appbar.Content title="Active Jobs" />
      </Appbar.Header>

      {isLoading ? (
        <View style={styles.loadingContainer}>
          <ActivityIndicator size="large" color={theme.colors.primary} />
        </View>
      ) : errorMessage ? (
        <View style={styles.errorBanner}>
          <Text style={{ color: theme.colors.error }}>{errorMessage}</Text>
        </View>
      ) : jobs.length > 0 ? (
        <FlatList
          data={jobs}
          renderItem={({ item: job }) => (
            <Card
              onPress={() => router.push({ pathname: '/(app)/jobs/[id]', params: { id: job.id } })}
              style={styles.card}
            >
              <Card.Title
                title={job.category}
                subtitle={job.timeframe}
              />
              <Card.Content>
                <Text variant="bodyMedium" numberOfLines={2}>
                  {job.description}
                </Text>
              </Card.Content>
            </Card>
          )}
          keyExtractor={(job) => job.id}
          contentContainerStyle={styles.listContent}
          refreshControl={<RefreshControl refreshing={isRefreshing} onRefresh={() => fetchJobs(true)} />}
        />
      ) : (
        <View style={styles.emptyContainer}>
          <Text variant="headlineSmall" style={styles.emptyText}>
            You have no active jobs
          </Text>
          <Text variant="bodyMedium" style={styles.emptySubtext}>
            Browse available jobs in the Feed tab to get started
          </Text>
        </View>
      )}
    </View>
  )
}

const styles = StyleSheet.create({
  container: { flex: 1 },
  loadingContainer: { flex: 1, alignItems: 'center', justifyContent: 'center' },
  errorBanner: { paddingHorizontal: 24, paddingVertical: 16 },
  card: { marginHorizontal: 12, marginBottom: 12 },
  listContent: { paddingHorizontal: 12, paddingVertical: 16 },
  emptyContainer: { flex: 1, alignItems: 'center', justifyContent: 'center', paddingHorizontal: 24 },
  emptyText: { marginBottom: 8 },
  emptySubtext: { textAlign: 'center' },
})
```

**UI Notes:**
- Appbar.Header with just title (no back button — top-level tab)
- Card for each job showing category (title), timeframe (subtitle), description (content)
- Tap card navigates to `/app/jobs/[id]` (handled by Phase 9 plan 03 to show conditional CTA)
- Pull-to-refresh on FlatList
- Empty state when `jobs.length === 0`: "You have no active jobs" (friendly, per UI-SPEC patterns from Phase 8)
- Error banner for network/API failures
- Loading spinner on initial load

**Why this pattern:** Mirrors the proven Phase 8 feed tab structure — token loading, fetch callback with error handling, tab focus refresh for data freshness, conditional rendering for all states. Reuses Paper components and styling already familiar in the codebase.
